def self.config(username,
				password,
				account_id, 
				integrator_key,
				endpoint_server = "www",
				api_version = "3.0")
	$_signatory_CONFIG[:username] = username
	$_signatory_CONFIG[:password] = password
	$_signatory_CONFIG[:account_id] = account_id
	$_signatory_CONFIG[:integrator_key] = integrator_key
	$_signatory_CONFIG[:endpoint_server] = endpoint_server
	$_signatory_CONFIG[:api_version] = api_version
	$_signatory_CONFIG[:endpoint] = "https://#{endpoint_server}.docusign.net/api/#{api_version}/"

	return $_signatory_CONFIG
end
